AOTA_china_refresh_unification_stage = {
    remove_ideas = AOTA_china_claimant_regime
    remove_ideas = AOTA_china_provisional_authority
    remove_ideas = AOTA_china_coalition_government
    remove_ideas = AOTA_china_contested_hegemon
    remove_ideas = AOTA_china_reunified_unstable
    remove_ideas = AOTA_china_reconstruction_state

    clr_country_flag = AOTA_china_stage_claimant
    clr_country_flag = AOTA_china_stage_provisional
    clr_country_flag = AOTA_china_stage_coalition
    clr_country_flag = AOTA_china_stage_contested
    clr_country_flag = AOTA_china_stage_reunified

    if = {
        limit = { check_variable = { AOTA_china_legitimacy < 20 } }
        add_ideas = AOTA_china_claimant_regime
        set_country_flag = AOTA_china_stage_claimant
    }
    if = {
        limit = {
            check_variable = { AOTA_china_legitimacy >= 20 }
            check_variable = { AOTA_china_legitimacy < 40 }
        }
        add_ideas = AOTA_china_provisional_authority
        set_country_flag = AOTA_china_stage_provisional
    }
    if = {
        limit = {
            check_variable = { AOTA_china_legitimacy >= 40 }
            check_variable = { AOTA_china_legitimacy < 60 }
        }
        add_ideas = AOTA_china_coalition_government
        set_country_flag = AOTA_china_stage_coalition
    }
    if = {
        limit = {
            check_variable = { AOTA_china_legitimacy >= 60 }
            check_variable = { AOTA_china_legitimacy < 80 }
        }
        add_ideas = AOTA_china_contested_hegemon
        set_country_flag = AOTA_china_stage_contested
    }
    if = {
        limit = {
            check_variable = { AOTA_china_legitimacy >= 80 }
        }
        add_ideas = AOTA_china_reunified_unstable
        set_country_flag = AOTA_china_stage_reunified
    }
}

AOTA_china_gain_legitimacy_small = {
    add_to_variable = { AOTA_china_legitimacy = 3 }
    if = { limit = { check_variable = { AOTA_china_legitimacy > 100 } } set_variable = { AOTA_china_legitimacy = 100 } }
    if = { limit = { check_variable = { AOTA_china_legitimacy < 0 } } set_variable = { AOTA_china_legitimacy = 0 } }
    AOTA_china_refresh_unification_stage = yes
}

AOTA_china_gain_legitimacy_medium = {
    add_to_variable = { AOTA_china_legitimacy = 6 }
    if = { limit = { check_variable = { AOTA_china_legitimacy > 100 } } set_variable = { AOTA_china_legitimacy = 100 } }
    if = { limit = { check_variable = { AOTA_china_legitimacy < 0 } } set_variable = { AOTA_china_legitimacy = 0 } }
    AOTA_china_refresh_unification_stage = yes
}

AOTA_china_lose_legitimacy_small = {
    add_to_variable = { AOTA_china_legitimacy = -4 }
    if = { limit = { check_variable = { AOTA_china_legitimacy > 100 } } set_variable = { AOTA_china_legitimacy = 100 } }
    if = { limit = { check_variable = { AOTA_china_legitimacy < 0 } } set_variable = { AOTA_china_legitimacy = 0 } }
    AOTA_china_refresh_unification_stage = yes
}
